{
    fast;

    ll i,j,k,n,m;
    string a,b;

    if(n<m){
        a=string(m-n,'0')+a;
    }
    else if(m<n){
        b=string(n-m,'0')+b;
    }
    n=max(n,m);
    one[0]=(b[0]=='1');
    for(i=1;i<n;i++) one[i]=one[i-1]+(b[i]=='1');
    pw[0]=1;
    for(i=1;i<N;i++) pw[i]=pw[i-1]*2LL%mod;
    ll ans=0;
    for(i=0;i<n;i++){
        ll res=(a[i]=='1');
        res=res*pw[n-i-1]%mod;
        ans=(ans+res*one[i]%mod)%mod;
    }
    cout<<ans<<endl;
